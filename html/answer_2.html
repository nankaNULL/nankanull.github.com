#include < iostream >
using namespace std;
int main (){
    int n;
    cin>>n;
    if(n%2==0&&n%3==0)
        cout<< n <<" divisible by 2 and 3? YES"<< endl;
    else
        cout<< n <<" divisible by 2 and 3? NO"<< endl;
    if(n%2==0||n%3==0)
        cout<< n <<" divisible by 2 or 3? YES"<< endl;
    else
        cout<< n <<" divisible by 2 or 3? NO"<< endl;
    if((n%2==0&&n%3!=0)&&(n%2!=0&&n%3==0))
        cout<< n <<" divisible by 2 or 3, but not both? YES"<< endl;
    else
        cout<< n <<" divisible by 2 or 3, but not both? NO"<< endl;
    return 0;
}
